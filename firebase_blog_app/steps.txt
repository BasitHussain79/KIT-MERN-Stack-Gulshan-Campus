1. create a project in firebase
2. Register our blog app in firebase
3. npm init -y --> to create a package.json file
4. npm i firebase
5. Goto the fb settings copy the config code and paste on the top of the app.js
6. npm i -D webpack webpack-cli webpack-dev-server
7. create webpack.config.js file
8. add code in webpack.config.js file
	const path = require("path");

	module.exports = {
  		mode: "development",
  		entry: {
    		main: path.resolve(__dirname, "./src/app.js"),
  		},
  		output: {
    		path: path.resolve(__dirname, "dist"),
    		filename: "bundle.js",
  		},
  		watch: true,
	
	};
9. go to the package.json add script

	"dev": "webpack --mode development"
10. npm run dev
